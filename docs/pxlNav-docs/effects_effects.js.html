// pxlNav Includes -
//   Primary Effect's File Conduit
//     Kevin Edzenga, 2024
// -- -- --
// Effects consist of the Effect Manager and the Effect itself.
//   For the case of Particles -
//     The particle system is the Manager and the particles themselves are Shaders

/**
 * @file effects.js
 * @description Primary Effect's File Conduit for pxlNav. This module includes the Effect Manager and the Effect itself.
 * @module pxlNav/effects
 * 
 * @requires ./particles.js
 */


import { BillowSmoke, EmberWisps, FloatingDust, HeightMap } from './particles.js';

/**
 * @namespace pxlEffects
 * @property {Object} pxlParticles - Collection of particle effects.
 * @property {Object} pxlParticles.BillowSmoke - BillowSmoke particle effect.
 * @property {Object} pxlParticles.EmberWisps - EmberWisps particle effect.
 * @property {Object} pxlParticles.FloatingDust - FloatingDust particle effect.
 * @property {Object} pxlParticles.HeightMap - HeightMap particle effect.
 */
const pxlEffects = { 
  "pxlParticles" : { BillowSmoke, EmberWisps, FloatingDust, HeightMap },
};

export { pxlEffects };
